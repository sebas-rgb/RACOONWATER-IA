#include <WiFi.h>
#include <WiFiClientSecure.h>
#include <PubSubClient.h>
#include <DHT.h>

// Pines sensores simulados
#define PIN_PH 34
#define PIN_TURBIDEZ 35
#define PIN_DHT 15
#define DHTTYPE DHT22

// WiFi Wokwi
const char* ssid = "Wokwi-GUEST";
const char* wifi_password = "";

// MQTT HiveMQ Cloud
const char* mqtt_server = "f446f631d7fd4f6883bf14f45054806f.s1.eu.hivemq.cloud";
const int mqtt_port = 8883;
const char* topic = "aquasense/sensores/lecturas";
const char* mqtt_user = "ESP32";
const char* mqtt_password = "Pruebaesp32";




WiFiClientSecure espClient;
PubSubClient client(espClient);
DHT dht(PIN_DHT, DHTTYPE);

void conectarWiFi() {
   Serial.print("Conectando a WiFi");
  WiFi.begin(ssid, wifi_password);

  int intentos = 0;

  while (WiFi.status() != WL_CONNECTED && intentos < 20) {
    delay(500);
    Serial.print(".");
    intentos++;
  }

  if (WiFi.status() == WL_CONNECTED) {
    Serial.println("\nWiFi conectado");
    Serial.print("IP: ");
    Serial.println(WiFi.localIP());
  } else {
    Serial.println("\nNo se pudo conectar a WiFi");
  }
}

void conectarMQTT() {
  while (!client.connected()) {
    Serial.print("Conectando a MQTT... ");

    if (client.connect("ESP32_AquaSense", mqtt_user, mqtt_password)) {
      Serial.println("conectado");
    } else {
      Serial.print("fallo, rc=");
      Serial.println(client.state());
      Serial.println("Reintentando en 3 segundos...");
      delay(3000);
    }
  }
}

float leerPH() {
  int rawPH = analogRead(PIN_PH);
  float ph = (rawPH / 4095.0) * 14.0;
  return ph;
}

float leerTurbidez() {
  int rawTurbidez = analogRead(PIN_TURBIDEZ);
  float turbidez = (rawTurbidez / 4095.0) * 1000.0;
  return turbidez;
}

float leerTemperatura() {
  float temperatura = dht.readTemperature();

  if (isnan(temperatura)) {
    Serial.println("Error leyendo temperatura");
    return 0.0;
  }

  return temperatura;
}

void setup() {
  Serial.begin(115200);
  delay(1000);

  Serial.println("=================================");
  Serial.println("Iniciando AquaSense IA - ESP32");
  Serial.println("=================================");

  pinMode(PIN_PH, INPUT);
  pinMode(PIN_TURBIDEZ, INPUT);

  dht.begin();

  conectarWiFi();

  espClient.setInsecure(); 
  client.setServer(mqtt_server, mqtt_port);

  Serial.println("Configuracion MQTT cargada");
}

void loop() {
  Serial.println("Loop ejecutándose...");
delay(1000);
  if (!client.connected()) {
    conectarMQTT();
  }

  client.loop();

  float ph = leerPH();
  float turbidez = leerTurbidez();
  float temperatura = leerTemperatura();

  String payload = "{";
  payload += "\"sensor_id\":\"sensor_001\",";
  payload += "\"zona_id\":1,";
  payload += "\"ph\":" + String(ph, 2) + ",";
  payload += "\"turbidez\":" + String(turbidez, 2) + ",";
  payload += "\"temperatura\":" + String(temperatura, 2);
  payload += "}";

  client.publish(topic, payload.c_str());

  Serial.println("Dato enviado por MQTT:");
  Serial.println(payload);
  

  delay(5000);
}
